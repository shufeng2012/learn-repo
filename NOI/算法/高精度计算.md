# 加法

```cpp
string add(string a,string b){
    reverse(a.begin(),a.end());
    reverse(b.begin(),b.end());
    string c;
    int i=0,x=0;
    while((i<=a.size())||(i<=b.size())){
        c[i]=(a[i]-'0')+(b[i]-'0')+x;
        if(c[i]>=10){
            x=c[i]%10;
            c[i]%=10;
        }
        c[i]+='0';
    }
    if(x!=0){
        char t=x+'0';
        c+=t;
    }
    reverse(c.begin(),c.end());
    return c;
}
```

# 减法

```cpp
string sub(string a,string b){
    reverse(a.begin(),a.end());
    reverse(b.begin(),b.end());
    string c;
    if(a.size()<b.size()||a<b)
        swap(a,b);
    int i;
    for(i=0;i<=b.size;i++){
        c[i]=a[i]-b[i];
        if(c[i]<0){
            a[i+1]--;
            c[i]+=10;
        }
        c[i]+='0';
    }
    if(i<a.size())
        while(i<=a.size()){
            if(a[i]<'0'){
                a[i+1]--;
                a[i]+=10;
            }
            c[i]=a[i];
        }
    reverse(c.begin(),c.end());
    return c;
}
```
